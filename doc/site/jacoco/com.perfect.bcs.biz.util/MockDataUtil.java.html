    public static String getName() {

        List<String> surnameList = getSurnameList();
        List<String> girl = getFemaleList();
        List<String> boy = getMaleList();

        // 获得一个随机的姓氏
        int index = RandomUtils.nextInt(0, surnameList.size());
        String name = surnameList.get(index);

        // 可以根据这个数设置产生的男女比例
        int i = RandomUtils.nextInt(0, 3);
        if (i == 2) {
            int j = RandomUtils.nextInt(0, girl.size() - 1);
            if (j % 2 == 0) {
                // 女
                name = name + girl.get(j) + girl.get(j + 1);
            } else {
                name = name + girl.get(j);
            }
        } else {
            int j = RandomUtils.nextInt(0, boy.size() - 1);
            // 男
            if (j % 2 == 0) {
                name = name + boy.get(j) + boy.get(j + 1);
            } else {
                name = name + boy.get(j);
            }
        }

        return name;
    }
